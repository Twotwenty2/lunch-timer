<script>

    let currentDuration = 30;

    let targetTime = null;

    let countdownInterval = null;

    let timerRunning = false;


    // ── Service Worker Registration ──────────────────────────────────────

    if ('serviceWorker' in navigator) {

        window.addEventListener('load', () => {

            navigator.serviceWorker.register('/sw.js', { scope: '/' })

                .then(reg => {

                    console.log('SW registered, scope:', reg.scope);

                    reg.addEventListener('updatefound', () => {

                        const newWorker = reg.installing;

                        newWorker.addEventListener('statechange', () => {

                            if (newWorker.state === 'installed' && navigator.serviceWorker.controller) {

                                console.log('New version available – reload to update.');

                            }

                        });

                    });

                })

                .catch(err => console.warn('SW registration failed:', err));

        });

    }


    window.onload = function() {

        resetToNow();

        updateGlobalTimes();

        setInterval(updateGlobalTimes, 30000);

        document.addEventListener('click', (e) => {

            const popup = document.getElementById('timezonePopup');

            if (popup && !document.getElementById('tzTrigger').contains(e.target)) popup.style.display = 'none';

        });

    };


    function togglePopup() {

        const p = document.getElementById('timezonePopup');

        p.style.display = (p.style.display === 'block') ? 'none' : 'block';

    }


    function updateGlobalTimes() {

        const zones = { 'tz-et': 'America/New_York', 'tz-ct': 'America/Chicago', 'tz-pt': 'America/Los_Angeles', 'tz-lon': 'Europe/London', 'tz-tky': 'Asia/Tokyo', 'tz-syd': 'Australia/Sydney' };

        for (const [id, zone] of Object.entries(zones)) {

            const el = document.getElementById(id);

            if(el) el.innerText = new Date().toLocaleTimeString('en-US', { timeZone: zone, hour: '2-digit', minute: '2-digit', hour12: true });

        }

    }


    function resetToNow() {

        const d = new Date();

        document.getElementById('startTime').value = `${String(d.getHours()).padStart(2, '0')}:${String(d.getMinutes()).padStart(2, '0')}`;

        calculatePreview();

    }


    function setMode(m) {

        currentDuration = m;

        document.querySelectorAll('.btn-choice').forEach(b => b.classList.remove('active'));

        event.target.classList.add('active');

        if (timerRunning) {

            const val = document.getElementById('startTime').value;

            const walkActive = document.getElementById('walkMode').checked;

            let [h, min] = val.split(':').map(Number);

            let adjust = currentDuration - (walkActive ? 2 : 0);

            let now = new Date();

            targetTime = new Date(now.getFullYear(), now.getMonth(), now.getDate(), h, min + adjust);

            runCountdown();

        } else {

            calculatePreview();

        }

    }


    function calculatePreview() {

        const val = document.getElementById('startTime').value;

        const walkActive = document.getElementById('walkMode').checked;

        document.getElementById('bufferLabel').style.color = walkActive ? "var(--buffer-color)" : "var(--text-light)";

        if (!val || timerRunning) return;

        let [h, m] = val.split(':').map(Number);

        let adjust = currentDuration - (walkActive ? 2 : 0);

        let now = new Date();

        let previewTarget = new Date(now.getFullYear(), now.getMonth(), now.getDate(), h, m + adjust);

        const ampm = previewTarget.getHours() >= 12 ? 'PM' : 'AM';

        const displayH = previewTarget.getHours() % 12 || 12;

        const displayM = String(previewTarget.getMinutes()).padStart(2, '0');

        document.getElementById('finalTime').innerHTML = `${displayH}:${displayM}<span style="font-size:1.5rem; margin-left:5px;">${ampm}</span>`;

        document.getElementById('statusLabel').innerText = walkActive ? "Will leave for desk at" : "Will return to desk at";

        document.getElementById('countdown').innerText = `Duration: ${currentDuration}m`;

    }


    function handleMainAction() {

        const btn = document.getElementById('mainActionBtn');

        if (!timerRunning) {

            startTimer();

            btn.innerText = "Reset Timer";

            btn.classList.add('reset-mode');

        } else {

            stopTimer();

            btn.innerText = "Start Timer";

            btn.classList.remove('reset-mode');

        }

    }


    function startTimer() {

        timerRunning = true;

        const val = document.getElementById('startTime').value;

        const walkActive = document.getElementById('walkMode').checked;

        let [h, m] = val.split(':').map(Number);

        let adjust = currentDuration - (walkActive ? 2 : 0);

        let now = new Date();

        targetTime = new Date(now.getFullYear(), now.getMonth(), now.getDate(), h, m + adjust);

        document.getElementById('statusLabel').innerText = walkActive ? "Leave for desk at" : "Return to desk at";

        runCountdown();

    }


    function stopTimer() {

        timerRunning = false;

        if (countdownInterval) clearTimeout(countdownInterval);

        countdownInterval = null;

        document.getElementById('stickMan').style.display = 'none';

        document.getElementById('finalTime').style.color = "white";

        calculatePreview();

    }


    function runCountdown() {

        if (countdownInterval) clearTimeout(countdownInterval);

        countdownInterval = null;


        function tick() {

            if (!timerRunning) return;


            let now = new Date().getTime();

            let distance = targetTime.getTime() - now;


            if (distance <= 0) {

                document.getElementById('countdown').innerText = "RETURN NOW!";

                document.getElementById('countdown').style.color = "var(--alert-red)";

                document.getElementById('finalTime').style.color = "var(--alert-red)";

                document.getElementById('stickMan').style.display = 'block';

                document.getElementById('stickMan').style.animation = 'shake 0.5s infinite';

                return;

            }


            if (distance <= 120000) {

                document.getElementById('finalTime').style.color = "var(--buffer-color)";

                document.getElementById('stickMan').style.display = 'block';

            }


            let minutes = Math.floor((distance % (1000 * 60 * 60)) / (1000 * 60));

            let seconds = Math.floor((distance % (1000 * 60)) / 1000);

            document.getElementById('countdown').innerText = `Time left: ${String(minutes).padStart(2, '0')}:${String(seconds).padStart(2, '0')}`;

            document.getElementById('countdown').style.color = "var(--accent-color)";


            // Schedule next tick exactly on the next whole-second boundary

            let msUntilNextSecond = distance % 1000;

            if (msUntilNextSecond === 0) msUntilNextSecond = 1000;

            countdownInterval = setTimeout(tick, msUntilNextSecond);

        }


        // Align the very first tick to the next whole-second boundary

        // so the counter is in phase from the start and never skips.

        let now = new Date().getTime();

        let distance = targetTime.getTime() - now;

        let msUntilNextSecond = distance % 1000;

        if (msUntilNextSecond === 0) msUntilNextSecond = 1000;

        countdownInterval = setTimeout(tick, msUntilNextSecond);

    }

</script>
