<script>

    let currentDuration = 30;


    window.onload = resetToNow;


    function resetToNow() {

        const d = new Date();

        document.getElementById('startTime').value = `${String(d.getHours()).padStart(2, '0')}:${String(d.getMinutes()).padStart(2, '0')}`;

        updateTime();

    }


    function setMode(m) {

        currentDuration = m;

        document.querySelectorAll('.btn-choice').forEach(b => b.classList.remove('active'));

        document.querySelector(`.btn-choice[data-min="${m}"]`).classList.add('active');

        updateTime();

    }


    function updateTime() {

        const val = document.getElementById('startTime').value;

        if (!val) return;


        let [h, m] = val.split(':').map(Number);

        let adjust = currentDuration;

        let log = `${currentDuration}m break`;


        if (document.getElementById('walkMode').checked) {

            adjust -= 2;

            log += ` - 2m walk`;

            document.getElementById('statusLabel').innerText = "Leave for desk at";

            document.getElementById('finalTime').style.color = "var(--buffer-color)";

        } else {

            document.getElementById('statusLabel').innerText = "Return to desk at";

            document.getElementById('finalTime').style.color = "white";

        }


        let totalMins = (h * 60) + m + adjust;

        if (totalMins < 0) totalMins += 1440;


        let resH = Math.floor((totalMins / 60) % 24);

        let resM = Math.floor(totalMins % 60);


        const ampm = resH >= 12 ? 'PM' : 'AM';

        const displayH = resH % 12 || 12;

        const displayM = String(resM).padStart(2, '0');


        document.getElementById('finalTime').innerText = `${displayH}:${displayM} ${ampm}`;

        document.getElementById('calcLog').innerText = `(${log})`;

    }

</script>
